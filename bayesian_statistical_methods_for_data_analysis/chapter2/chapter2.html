Box-Cox変換

library(car)
x <- read.table("../r3/dataset/PDA-F.txt", header=T)
hist(x[,1])

# Box-Cox変換
# powerTransformをかけてλを推定
# bxPowerで変換
summary(p1 <- powerTransform(x[,1]))
## bcPower Transformation to Normality 
## 
##        Est.Power Std.Err. Wald Lower Bound Wald Upper Bound
## x[, 1]   -0.2432   0.0483          -0.3378          -0.1486
## 
## Likelihood ratio tests about transformation parameters
##                             LRT df         pval
## LR test, lambda = (0)  25.72705  1 3.932757e-07
## LR test, lambda = (1) 743.16266  1 0.000000e+00
hist(bcPower(x[,1], p1$roundlam)) # 変換前, 変換後
